      startTime: Date.now(),

    };

    if (this.onStateChangeCallback) {
      this.onStateChangeCallback(this.state);
    }

  /**

   */
  private spreadTick(): void {
    if (!this.gridData || this.state.type !== 'running') return;

    const newGrid = [...this.gridData.grid];
    const { width, height } = this.gridData;

    // ÐŸÑ€Ð¾Ñ…Ð¾Ð´Ð¸Ð¼ Ð¿Ð¾ Ð²ÑÐµÐ¼ ÐºÐ»ÐµÑ‚ÐºÐ°Ð¼
    for (let y = 0; y < height; y++) {

    this.gridData.grid = newGrid;

    // Ð£Ð²ÐµÐ»Ð¸Ñ‡Ð¸Ð²Ð°ÐµÐ¼ ÑÑ‡Ñ‘Ñ‚Ñ‡Ð¸Ðº Ñ‚Ð¸ÐºÐ°
    if (this.state.type === 'running') {
      this.state = {
        ...this.state,
        tick: this.state.tick + 1
      };
    }

    // Ð£Ð²ÐµÐ´Ð¾Ð¼Ð»ÑÐµÐ¼ Ð¾Ð± Ð¾Ð±Ð½Ð¾Ð²Ð»ÐµÐ½Ð¸Ð¸
    if (this.onGridUpdateCallback) {
      this.onGridUpdateCallback(this.gridData.grid);
    }

    // ÐŸÑ€Ð¾Ð²ÐµÑ€ÑÐµÐ¼ Ð¿Ð¾Ð±ÐµÐ´Ñƒ
    this.checkWinCondition();
  }

  /**
   * Ð Ð°ÑÐ¿Ñ€Ð¾ÑÑ‚Ñ€Ð°Ð½Ð¸Ñ‚ÑŒ Ð²Ð¸Ñ€ÑƒÑ Ð¸Ð· ÐºÐ»ÐµÑ‚ÐºÐ¸
   */
  private spreadVirus(
    grid: number[],
    x: number,
    y: number,
    virusType: number,
    width: number,
    height: number
  ): void {
    // ÐŸÐ°Ñ€Ð°Ð¼ÐµÑ‚Ñ€Ñ‹ Ð°ÐºÑ‚Ð¸Ð²Ð½Ð¾Ð³Ð¾ Ð²Ð¸Ñ€ÑƒÑÐ°
    const params = virusType === 1 ? this.paramsA : this.paramsB;
    if (!params) return;

    // Ð¡Ð¾ÑÐµÐ´Ð½Ð¸Ðµ ÐºÐ»ÐµÑ‚ÐºÐ¸ (8 Ð½Ð°Ð¿Ñ€Ð°Ð²Ð»ÐµÐ½Ð¸Ð¹)
    const neighbors = [
      { dx: -1, dy: 0 },  // Ð»ÐµÐ²Ð¾
      { dx: 1, dy: 0 },   // Ð¿Ñ€Ð°Ð²Ð¾
      { dx: 0, dy: -1 },  // Ð²ÐµÑ€Ñ…
      { dx: 0, dy: 1 },   // Ð½Ð¸Ð·
      { dx: -1, dy: -1 }, // Ð²ÐµÑ€Ñ…-Ð»ÐµÐ²Ð¾
      { dx: 1, dy: -1 },  // Ð²ÐµÑ€Ñ…-Ð¿Ñ€Ð°Ð²Ð¾
      { dx: -1, dy: 1 },  // Ð½Ð¸Ð·-Ð»ÐµÐ²Ð¾
      { dx: 1, dy: 1 },   // Ð½Ð¸Ð·-Ð¿Ñ€Ð°Ð²Ð¾
    ];

    // Ð¨Ð°Ð½Ñ Ñ€Ð°ÑÐ¿Ñ€Ð¾ÑÑ‚Ñ€Ð°Ð½ÐµÐ½Ð¸Ñ (Ð·Ð°Ð²Ð¸ÑÐ¸Ñ‚ Ð¾Ñ‚ speed + reproduction)
    const spreadChance = (params.speed + params.reproduction) / 20; // 0.0 - 1.0

    // ÐšÐ¾Ð»Ð¸Ñ‡ÐµÑÑ‚Ð²Ð¾ Ð¿Ð¾Ð¿Ñ‹Ñ‚Ð¾Ðº Ñ€Ð°ÑÐ¿Ñ€Ð¾ÑÑ‚Ñ€Ð°Ð½ÐµÐ½Ð¸Ñ (Ð·Ð°Ð²Ð¸ÑÐ¸Ñ‚ Ð¾Ñ‚ contagiousness)
    const spreadAttempts = 1 + Math.floor(params.contagiousness / 5);

    for (let attempt = 0; attempt < spreadAttempts; attempt++) {
      // Ð’Ñ‹Ð±Ð¸Ñ€Ð°ÐµÐ¼ ÑÐ»ÑƒÑ‡Ð°Ð¹Ð½Ð¾Ð³Ð¾ ÑÐ¾ÑÐµÐ´Ð°
      const neighbor = neighbors[Math.floor(Math.random() * neighbors.length)];
      const nx = x + neighbor.dx;
      const ny = y + neighbor.dy;

      // ÐŸÑ€Ð¾Ð²ÐµÑ€ÑÐµÐ¼ Ð³Ñ€Ð°Ð½Ð¸Ñ†Ñ‹
      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;

      const nIdx = ny * width + nx;
      const neighborCell = grid[nIdx];

      // ÐŸÑƒÑÑ‚Ð°Ñ ÐºÐ»ÐµÑ‚ÐºÐ° - Ð·Ð°Ñ…Ð²Ð°Ñ‚Ñ‹Ð²Ð°ÐµÐ¼
      if (neighborCell === 0) {
        if (Math.random() < spreadChance) {
          grid[nIdx] = virusType;
        }
      }
      // ÐšÐ»ÐµÑ‚ÐºÐ° Ð²Ñ€Ð°Ð³Ð° - Ð°Ñ‚Ð°ÐºÑƒÐµÐ¼
      else if (neighborCell !== virusType) {
        this.attackCell(grid, nIdx, virusType, neighborCell, params);
      }
    }
  }

  /**
   * ÐÑ‚Ð°ÐºÐ° ÐºÐ»ÐµÑ‚ÐºÐ¸ Ð²Ñ€Ð°Ð³Ð°
   */
  private attackCell(
    grid: number[],
    idx: number,
    attackerType: number,
    defenderType: number,
    attackerParams: VirusParams
  ): void {
    // ÐŸÐ°Ñ€Ð°Ð¼ÐµÑ‚Ñ€Ñ‹ Ð·Ð°Ñ‰Ð¸Ñ‚Ð½Ð¸ÐºÐ°
    const defenderParams = defenderType === 1 ? this.paramsA : this.paramsB;
    if (!defenderParams) return;

    // Ð¡Ð¸Ð»Ð° Ð°Ñ‚Ð°ÐºÐ¸ (aggression + virulence + lethality)
    const attackPower = attackerParams.aggression + attackerParams.virulence + attackerParams.lethality;

    // Ð¡Ð¸Ð»Ð° Ð·Ð°Ñ‰Ð¸Ñ‚Ñ‹ (defense + resilience)
    const defensePower = defenderParams.defense + defenderParams.resilience;

    // Ð¨Ð°Ð½Ñ Ð·Ð°Ñ…Ð²Ð°Ñ‚Ð°
    const captureChance = 0.3 + (attackPower - defensePower) / 30; // Ð‘Ð°Ð·Ð¾Ð²Ñ‹Ð¹ 30% + Ð¼Ð¾Ð´Ð¸Ñ„Ð¸ÐºÐ°Ñ‚Ð¾Ñ€Ñ‹

    if (Math.random() < captureChance) {
      // Ð—Ð°Ñ…Ð²Ð°Ñ‚ ÐºÐ»ÐµÑ‚ÐºÐ¸
      grid[idx] = attackerType;
    }
  }

  /**
   * ÐŸÑ€Ð¾Ð²ÐµÑ€Ð¸Ñ‚ÑŒ ÑƒÑÐ»Ð¾Ð²Ð¸Ðµ Ð¿Ð¾Ð±ÐµÐ´Ñ‹
   */
  private checkWinCondition(): void {
    if (!this.gridData) return;

    let countA = 0;
    let countB = 0;

    for (const cell of this.gridData.grid) {
      if (cell === 1) countA++;
      else if (cell === 2) countB++;
    }

    const total = countA + countB;
    if (total === 0) return;

    const percentA = (countA / total) * 100;
    const percentB = (countB / total) * 100;

    // ÐŸÐ¾Ð±ÐµÐ´Ð° ÐµÑÐ»Ð¸ 99% Ñ‚ÐµÑ€Ñ€Ð¸Ñ‚Ð¾Ñ€Ð¸Ð¸
    if (percentA >= 99) {
      this.endBattle('A', countA, countB);
    } else if (percentB >= 99) {
      this.endBattle('B', countA, countB);
    }
    // Ð˜Ð»Ð¸ ÐµÑÐ»Ð¸ Ð¾Ð´Ð¸Ð½ Ð²Ð¸Ñ€ÑƒÑ Ð¿Ð¾Ð»Ð½Ð¾ÑÑ‚ÑŒÑŽ ÑƒÐ½Ð¸Ñ‡Ñ‚Ð¾Ð¶ÐµÐ½
    else if (countA === 0 && countB > 0) {
      this.endBattle('B', countA, countB);
    } else if (countB === 0 && countA > 0) {
      this.endBattle('A', countA, countB);
    }
  }

  /**
   * Ð—Ð°Ð²ÐµÑ€ÑˆÐ¸Ñ‚ÑŒ Ð±Ð¸Ñ‚Ð²Ñƒ

      clearInterval(this.spreadInterval);
      this.spreadInterval = null;
    }

    this.state = {
      type: 'ended',
      winner,
      virusACount: countA,
      virusBCount: countB
    };

    if (this.onStateChangeCallback) {
      this.onStateChangeCallback(this.state);
    }
  }

  }

    this.endBattle(data.winner as 'A' | 'B', data.virusACount, data.virusBCount);
  }
